	\begin{frame}[fragile]{\textbf{\textit{\textit{\stepcounter{subslide}\slidenum. Cách chèn các kí tự đặc biệt trong \LaTeX}}}}
		\large Trong \LaTeX, ký tự đặc biệt là những ký tự không thể nhập trực tiếp như văn bản bình thường, vì chúng có ý nghĩa riêng trong hệ thống.
		\begin{block}{\textbf{\textit{Ví dụ:}}}
			\begin{itemize}
				\item \textbf{\$} dùng để mở chế độ toán
				\item \textbf{\%} dùng để comment
				\item \textbf{\_} dùng để viết chỉ số dưới
			\end{itemize}
			\textbf{$\to$ Vì vậy, nếu muốn hiển thị các ký tự này, ta có cách viết đặc biệt.}
		\end{block}
	\end{frame}
    \begin{frame}{\textbf{\textit{\stepcounter{subslide}\slidenum. Dùng dấu \textbackslash\ (escape) để chèn ký tự đặc biệt}}}
    	\begin{block}{\textbf{Một số ký tự phải thêm \textbackslash\ phía trước:}}
    		\begin{table}[h]
    			\begin{tabular}{|c|c|}
    				\hline
    				\rowcolor{red!80}
    				\color{white}\textbf{Ký tự} & \color{white}\textbf{Cách viết trong} \textbf{\LaTeX}\\
    				\hline
    				\# & \textbackslash\#\\
    				\hline
    				\$ & \textbackslash\$\\
    				\hline
    				\% & \textbackslash\%\\
    				\hline
    				\& & \textbackslash\&\\
    				\hline
    				\_ & \textbackslash\_\\
    				\hline
    				\{\} & \textbackslash\{ \textbackslash\}\\
    				\hline
    				\^{}&\textbackslash\textasciicircum\{ \}\\
    				\hline
    				\~{}&\textbackslash\~{}\{ \}\\
    				\hline
    				\textbackslash\ & \texttt{\textbackslash textbackslash}\\
    				\hline
    			\end{tabular}
    		\end{table}
    	\end{block}
		\textit{\textbf{Ví dụ:} Giá tiền là \textbf{\texttt{\textbackslash\$100}} và giảm \textbf{\texttt{20\textbackslash\%}}} $\to$ Giá tiền là \textbf{\$100} và giảm \textbf{20\%.}
	\end{frame}

	\begin{frame}[fragile]{\textbf{\textit{2. Cách chèn các kí tự đặc biệt trong \LaTeX}}}
		\large Trong \LaTeX, ký tự đặc biệt là những ký tự không thể nhập trực tiếp như văn bản bình thường, vì chúng có ý nghĩa riêng trong hệ thống.
		\begin{block}{\textbf{\textit{Ví dụ:}}}
			\begin{itemize}
				\item \textbf{\$} dùng để mở chế độ toán
				\item \textbf{\%} dùng để comment
				\item \textbf{\_} dùng để viết chỉ số dưới
			\end{itemize}
			\textbf{$\to$ Vì vậy, nếu muốn hiển thị các ký tự này, ta có cách viết đặc biệt.}
		\end{block}
	\end{frame}
    \begin{frame}{\textbf{2.1 Dùng dấu \textbackslash\ (escape) để chèn ký tự đặc biệt}}
    	\textbf{Một số ký tự phải thêm \textbackslash\ phía trước:}
		\begin{table}[h]
			\begin{tabular}{|c|c|}
				\hline
				\rowcolor{red!80}
				\color{white}\textbf{Ký tự} & \color{white}\textbf{Cách viết trong} \textbf{\LaTeX}\\
				\hline
				\# & \textbackslash\#\\
				\hline
				\$ & \textbackslash\$\\
				\hline
				\% & \textbackslash\%\\
				\hline
				\& & \textbackslash\&\\
				\hline
				\_ & \textbackslash\_\\
				\hline
				\{\} & \textbackslash\{ \textbackslash\}\\
				\hline
				\^{}&\textbackslash\textasciicircum\{ \}\\
				\hline
				\~{}&\textbackslash\~{}\{ \}\\
				\hline
				\textbackslash\ & \texttt{\textbackslash textbackslash}\\
				\hline
			\end{tabular}
		\end{table}
		\textit{\textbf{Ví dụ:} Giá tiền là \textbf{\texttt{\textbackslash\$100}} và giảm \textbf{\texttt{20\textbackslash\%}}} $\to$ Giá tiền là \textbf{\$100} và giảm \textbf{20\%.}
	\end{frame}
	\begin{frame}{\textbf{2.2 Dùng lệnh có sẵn (symbols)}}
		\textbf{Các kí hiệu phổ biến:}
		\begin{table}[h]
			\begin{tabular}{|c|c|}
				\hline
				\rowcolor{red!80}
				\color{white}\textbf{Kí hiệu} &\color{white}\textbf{Cách viết trong \LaTeX}\\
				\hline
				\copyright & \texttt{\textbackslash copyright}\\
				\hline
				\textregistered & \texttt{\textbackslash textregistered}\\
				\hline
				\textdegree & \texttt{\textbackslash textdegree}\\
				\hline
				\dag & \texttt{\textbackslash dag}\\
				\hline
				\ddag & \texttt{\textbackslash ddag}\\
				\hline
				$\alpha, \beta, \gamma$ & \texttt{\textbackslash alpha}, \texttt{\textbackslash beta}, \texttt{\textbackslash gamma}\\
				\hline
				$\infty$ & \texttt{\textbackslash infty}\\
				\hline
				$\sum$ & \texttt{\textbackslash sum}\\
				\hline
				$\int$ & \texttt{\textbackslash int}\\
				\hline
				$\leq, \geq$ & \texttt{\textbackslash leq},\texttt{\textbackslash geq}\\
				\hline
			\end{tabular}
		\end{table}
		\textbf{Ví dụ:} \textit{Nhiệt độ là 30\textbf{\texttt{\textbackslash textdegree C}}} $\to$ Nhiệt độ là \textbf{30\textdegree C.}
	\end{frame}

	\begin{frame}{\textbf{\textit{Mục 3}}}
		\centering
		\Huge
		\textcolor{blue!60}{\textbf{Chèn các biểu tượng hình ảnh trong \LaTeX}}
	\end{frame}
	\begin{frame}[fragile]{\textbf{\textit{Ký hiệu toán học}}}%fragile cho phép sử dụng verb trong beamer
		\begin{block}{Khai báo:}
			\verb|Môi trường \usepackage{amssymb}|
		\end{block}
		\pause
		\begin{block}{Một vài kí tự toán học:}
			\begin{table}
				\begin{tabular}{|c|l|}
					\hline
				Ký hiệu & Công thức\\
				\hline
				$\checkmark$ & \verb|\checkmark|\\
				\hline
				$\times$ & \verb|\times|\\
				\hline
				$\star$ & \verb|\star|\\
				\hline
				$\dagger$ & \verb|\dagger|\\
				\hline
				\end{tabular}
			\end{table}
		\end{block}
		\pause
		\begin{alertblock}{Lưu ý:}
			Các kí hiệu toán học phải được đặt trong math mode (chế độ toán) như là \verb|$...$, \(...\), \[...\],...|
		\end{alertblock}
		\pause
	\end{frame}
	\begin{frame}[fragile]{\textbf{\textit{Biểu tượng dạng icon}}}
		\begin{block}{Khai báo:}
			\verb|Môi trường \usepackage{pifont}|
		\end{block}
		\pause
		\begin{block}{Cách dùng:}
			\verb|\ding{<số>}|
		\end{block}
		\pause
		\begin{exampleblock}{Ví dụ:}
			\ding{51}: \verb|\ding{51}|, \ding{55}: \verb|\ding{55}|, \ding{60}: \verb|\ding{60}|,...
		\end{exampleblock}
	\end{frame}
	\begin{frame}[fragile]{\textbf{\textit{Icon hiện đại}}}
		\begin{block}{Khai báo:}
			\verb|Môi trường \usepackage{fontawesome}|
		\end{block}
		\pause
		\begin{block}{Một số icon hiện đại:}
			\pause
			\begin{table}
				\begin{tabular}{|c|l|}
					\hline
					Icon & Công thức\\
					\hline
					\faHeart & \verb|\faHeart|\\
					\hline
					\faStar & \verb|\faStar|\\
					\hline
					\faUser & \verb|\faUser|\\
					\hline
					\faEnvelope & \verb|\faEnvelope|\\
					\hline
				\end{tabular}
			\end{table}
		\end{block}
	\end{frame}
	\begin{frame}[fragile]{\textbf{\textit{Biểu tượng thực tế}}}
		\begin{block}{Khai báo:}
			\verb|Môi trường usepackage{marvosym}|
		\end{block}
		\pause
		\begin{exampleblock}{Ví dụ: }
			\Telefon: \verb|\Telefon|, \Letter: \verb|\Letter|, \EUR: \verb|\EUR|, \Lightning: \verb|\Lightning| 
		\end{exampleblock}
		\pause
		\begin{block}{Mục đích:}
			Cung cấp các biểu tượng mang tính thực tế.\\
			Khác với pifont (thiên về kí hiệu), marvosym có icon như: điện thoại, email, tiền tệ, cảnh báo 
		\end{block}
	\end{frame}
